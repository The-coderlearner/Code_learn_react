const processExcelAndJson = require('./processExcelAndJson');

const excelFilePath = 'path/to/your/excel/file.xlsx'; // Path to your Excel file
const jsonFilePath = 'path/to/your/json/file.json'; // Path to your JSON data file
const mappingFilePath = 'path/to/your/mapping/file.json'; // Path to your JSON mapping file

processExcelAndJson(excelFilePath, jsonFilePath, mappingFilePath)
  .then(result => {
    console.log(result); // Will print true or false based on the presence of data
  })
  .catch(error => {
    console.error('Error processing Excel and JSON:', error);
  });
